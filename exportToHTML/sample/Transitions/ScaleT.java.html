18           if (!Objects.equals(setToX,null)){ 
19           scaleTransition.setToX(setToX); 
20           } 
21           if (!Objects.equals(setToY,null)) { 
22               scaleTransition.setToY(setToY); 
23           } 
24           if(!Objects.equals(setToZ,null)) {

26           }
