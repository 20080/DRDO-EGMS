226      public void printImage(Node node) { 
227   
228          Printer printer = Printer.getDefaultPrinter(); 
229          PageLayout pageLayout = printer.getDefaultPageLayout(); 
230          System.out.println("PageLayout: " + pageLayout); 
231   
232          // Printable area 
233          double pWidth = pageLayout.getPrintableWidth(); 
234          double pHeight = pageLayout.getPrintableHeight(); 
235          System.out.println("Printable area is " + pWidth + " width and " 
236                  + pHeight + " height."); 
237   
238          // Node's (Image) dimensions 
239          double nWidth = node.getBoundsInParent().getWidth(); 
240          double nHeight = node.getBoundsInParent().getHeight(); 
241          System.out.println("Node's dimensions are " + nWidth + " width and " 
242                  + nHeight + " height"); 
243   
244          // How much space is left? Or is the image to big? 
245          double widthLeft = pWidth - nWidth; 
246          double heightLeft = pHeight - nHeight; 
247          System.out.println("Width left: " + widthLeft 
248                  + " height left: " + heightLeft); 
249   
250          // scale the image to fit the page in width, height or both 
251          double scaleq = 0; 
252   
253          if (widthLeft < heightLeft) { 
254              scaleq = pWidth / nWidth; 
255          } else { 
256              scaleq = pHeight / nHeight; 
257          } 
258   
259          // preserve ratio (both values are the same) 
260          node.getTransforms().add(new Scale(0.75,0.75)); 
261   
262          // after scale you can check the size fit in the printable area 
263          double newWidth = node.getBoundsInParent().getWidth(); 
264          double newHeight = node.getBoundsInParent().getHeight(); 
265          System.out.println("New Node's dimensions: " + newWidth 
266                  + " width " + newHeight + " height"); 
267   
268          PrinterJob job = PrinterJob.createPrinterJob(); 
269          if (job != null) { 
270              boolean success = job.printPage(node); 
271              if (success) { 
272                  job.endJob(); 
273   
274              } 
275          } 
276      }
